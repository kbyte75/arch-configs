# ROUND IMAGE CORNERS USING IMAGEMAGICK THIS ARE PERCENTAGE %[fx:w*0.15]
magick input.png \
    \( +clone -alpha extract \
       -draw "fill black polygon 0,0 0,%[fx:w*0.15] %[fx:w*0.15],0 fill white circle %[fx:w*0.15],%[fx:w*0.15] %[fx:w*0.15],0" \
       \( +clone -flip \) -compose Multiply -composite \
       \( +clone -flop \) -compose Multiply -composite \
    \) -alpha off -compose CopyOpacity -composite output.png
